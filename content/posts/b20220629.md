724.Find Pivot Index

```Go
func pivotIndex(nums []int) int {
	n := len(nums)

	if n == 0 {
		return -1
	}
	if nums[1:] == nil {
		return 0
	}
	if nums[:n] == nil {
		return n - 1
	}
	for i := 0; i < n; i++ {
		if sum(nums[0:i]) == sum(nums[i+1:n]) {
			return i
		}
	}
	return -1
}

func sum(xs []int) int {
	var sum int
	for _, x := range xs {
		sum += x
	}
	return sum
}
```
